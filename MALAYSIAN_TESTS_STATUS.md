## Template Structure

Each test file should follow this structure:
```typescript
export const testName = {
  id: 'test-id',
  title: 'Test Title in Malay',
  description: 'Description in Malay',
  questions: [
    // 12 questions with 4 options each (type A, B, C, D)
  ],
  results: {
    A: { title: '...', description: '...' },
    B: { title: '...', description: '...' },
    C: { title: '...', description: '...' },
    D: { title: '...', description: '...' }
  },
  calculateResult: (answers: string[]) => {
    const counts = answers.reduce((acc, answer) => {
      acc[answer] = (acc[answer] || 0) + 1;
      return acc;
    }, {} as Record<string, number>);
    
    return Object.entries(counts).reduce((a, b) => counts[a[0]] > counts[b[0]] ? a : b)[0];
  }
};
```
